    <script>
        let orderCount = 0;

        // Initialize order number
        function initOrderNumber() {
            const savedCount = localStorage.getItem('orderCount');
            orderCount = savedCount ? parseInt(savedCount) : 0;
            updateOrderNumber();
        }

        // Update order number display
        function updateOrderNumber() {
            const orderNumberElement = document.getElementById('orderNumber');
            const formattedNumber = `Order #${orderCount.toString().padStart(4, '0')}`;
            orderNumberElement.textContent = formattedNumber;
        }

        // Handle product change to show/hide size field
        document.getElementById('product').addEventListener('change', function(e) {
            const sizeField = document.getElementById('size').parentElement;
            const product = e.target.value.toLowerCase();
            if (product.includes('mug') || product.includes('cup')) {
                sizeField.style.display = 'none';
                document.getElementById('size').removeAttribute('required');
            } else {
                sizeField.style.display = 'block';
                document.getElementById('size').setAttribute('required', 'required');
            }
        });

        // Handle delivery option change
        document.querySelectorAll('input[name="deliveryOption"]').forEach(radio => {
            radio.addEventListener('change', function() {
                const deliveryFields = document.getElementById('deliveryFields');
                const addressInput = document.getElementById('address');
                const phoneInput = document.getElementById('phone');

                if (this.value === 'paxi') {
                    deliveryFields.classList.add('active');
                    addressInput.setAttribute('required', 'required');
                    phoneInput.setAttribute('required', 'required');
                } else {
                    deliveryFields.classList.remove('active');
                    addressInput.removeAttribute('required');
                    phoneInput.removeAttribute('required');
                }
            });
        });

        // Handle form submission
        document.getElementById('orderForm').addEventListener('submit', function(e) {
            e.preventDefault();

            const customerName = document.getElementById('customerName').value;
            const product = document.getElementById('product').value;
            const color = document.getElementById('color').value;
            const size = document.getElementById('size').value;
            const quantity = document.getElementById('quantity').value;
            const deliveryOption = document.querySelector('input[name="deliveryOption"]:checked').value;
            const orderNum = orderCount.toString().padStart(4, '0');

            let deliveryInfo = '';
            if (deliveryOption === 'paxi') {
                const address = document.getElementById('address').value;
                const phone = document.getElementById('phone').value;
                deliveryInfo = `\nDelivery Address: ${address}\nContact Number: ${phone}\nDelivery Method: Paxi Courier`;
            } else {
                deliveryInfo = '\nDelivery Method: Collection';
            }

            // Format WhatsApp message
            const message = `New Order #${orderNum}

Customer: ${customerName}
Product: ${product}
Color: ${color}
${product.toLowerCase().includes('mug') || product.toLowerCase().includes('cup') ? '' : `Size: ${size}\n`}Quantity: ${quantity}${deliveryInfo}

Bank Details for Payment:
Bank: Capitec
Account Holder: RM Masilo
Account Number: 1694482896
Reference: Order #${orderNum}`;

            // Increment and save order count
            orderCount++;
            localStorage.setItem('orderCount', orderCount);
            updateOrderNumber();

            // Open WhatsApp with formatted message
            const whatsappLink = `https://wa.me/27733214017?text=${encodeURIComponent(message)}`;
            window.open(whatsappLink, '_blank');
        });

        // Add color preview to color select
        document.getElementById('color').addEventListener('change', function(e) {
            const selectedOption = e.target.options[e.target.selectedIndex];
            const color = selectedOption.getAttribute('data-color');
            if (color) {
                if (!document.querySelector('.color-preview')) {
                    const preview = document.createElement('span');
                    preview.className = 'color-preview';
                    e.target.parentNode.insertBefore(preview, e.target);
                }
                const preview = document.querySelector('.color-preview');
                preview.style.backgroundColor = color;
            } else {
                const preview = document.querySelector('.color-preview');
                if (preview) preview.remove();
            }
        });

        // Initialize on page load
        initOrderNumber();
    </script>
